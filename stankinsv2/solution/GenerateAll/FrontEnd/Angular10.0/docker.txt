@{
	var angular= "@angular";
}
FROM node:12.0
WORKDIR /app

RUN npm install -g @angular/cli@10.0.4

    && npm -v \
    && ng version

RUN ls -lh

ENTRYPOINT ["ng", "serve","--host", "0.0.0.0", "--port", "4200"]
	
EXPOSE 4200
